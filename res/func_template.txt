func %FuncName%(%Args%) %ResultType% {
    resp, err := http.Get("http://%URN%")

        fmt.Println(err)
        return ""
    }

        err = resp.Body.Close()
        if err != nil {
            fmt.Println(err)
        }
    }()
    body, err := ioutil.ReadAll(resp.Body)

}
